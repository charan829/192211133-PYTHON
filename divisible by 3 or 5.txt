def is_divisible_by_3_or_5(number):
    
    if number % 3 == 0 or number % 5 == 0:
        return True
    else:
        return False


num = int(input("enter the number:"))
if is_divisible_by_3_or_5(num):
    print(num, "is divisible by 3 or 5.")
else:
    print(num, "is not divisible by 3 or 5.")
