def fibonacci_numbers(n):
    fib_list = []
    a, b = 0, 1
    for _ in range(n):
        fib_list.append(a)
        a, b = b, a + b
    return fib_list

# Example usage:
n = int(input("Enter the number of Fibonacci numbers you want: "))
fibonacci_list = fibonacci_numbers(n)
print("First", n, "Fibonacci numbers:", fibonacci_list)
